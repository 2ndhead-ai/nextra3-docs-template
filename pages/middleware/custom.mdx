# Custom Middleware in Express.js

Custom middleware functions are a powerful way to extend Express.js with your own functionality. They allow you to modularize your application logic, separate concerns, and create reusable components for common operations such as authentication, logging, and error handling.

## Middleware Function Signature

Every Express.js middleware function follows a specific signature:

```javascript
function myMiddleware(req, res, next) {
  // Middleware code here
  next(); // Call next to pass control to the next middleware
}
```

The three parameters are:

1. **`req` (request)**: The HTTP request object containing information about the client's request, including headers, parameters, body, etc.

2. **`res` (response)**: The HTTP response object used to send responses to the client. It provides methods to set status codes, headers, and send data.

3. **`next`**: A function that, when called, passes control to the next middleware in the stack. If you don't call `next()`, the request processing pipeline stops, and the client waits for a response that may never come.

## Creating Simple Custom Middleware

Let's start with some basic examples of custom middleware.

### Simple Logger Middleware

```javascript
const loggerMiddleware = (req, res, next) => {
  console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);
  next(); // Don't forget to call next()!
};

// Apply middleware to all routes
app.use(loggerMiddleware);

// Or apply to specific routes
app.get('/users', loggerMiddleware, (req, res) => {
  res.json({ users: [] });
});
```

### Request Timer Middleware

```javascript
const requestTimer = (req, res, next) => {
  req.startTime = Date.now();
  
  // Override res.end to calculate and log the duration when the response is sent
  const originalEnd = res.end;
  res.end = function() {
    const duration = Date.now() - req.startTime;
    console.log(`Request to ${req.url} took ${duration}ms`);
    originalEnd.apply(res, arguments);
  };
  
  next();
};

app.use(requestTimer);
```

### Body Validator Middleware

```javascript
const validateUserBody = (req, res, next) => {
  const { name, email } = req.body;
  
  if (!name || !email) {
    return res.status(400).json({ error: 'Name and email are required' });
  }
  
  // If email format is invalid
  if (!/^\S+@\S+\.\S+$/.test(email)) {
    return res.status(400).json({ error: 'Invalid email format' });
  }
  
  // If validation passes, continue to the route handler
  next();
};

app.post('/users', validateUserBody, (req, res) => {
  // Create user logic here
});
```

## Implementation Examples

Now let's explore more practical and complex middleware implementations for common use cases.

### Logging Middleware

A more sophisticated logging middleware might include request details, response status codes, and execution time:

```javascript
const advancedLogger = (req, res, next) => {
  const startTime = Date.now();
  const requestId = Math.random().toString(36).substring(2, 15);
  
  // Store request ID for tracking
  req.requestId = requestId;
  
  // Log request details
  console.log(`[${requestId}] ${new Date().toISOString()} - ${req.method} ${req.url}`);
  console.log(`[${requestId}] Headers: ${JSON.stringify(req.headers)}`);
  
  if (req.body && Object.keys(req.body).length > 0) {
    console.log(`[${requestId}] Body: ${JSON.stringify(req.body)}`);
  }
  
  // Capture the original response methods
  const originalSend = res.send;
  const originalJson = res.json;
  const originalEnd = res.end;
  
  // Override send
  res.send = function(body) {
    console.log(`[${requestId}] Response body: ${body}`);
    return originalSend.call(this, body);
  };
  
  // Override json
  res.json = function(body) {
    console.log(`[${requestId}] Response JSON: ${JSON.stringify(body)}`);
    return originalJson.call(this, body);
  };
  
  // Override end to log when response finishes
  res.end = function() {
    const duration = Date.now() - startTime;
    console.log(`[${requestId}] Request completed - Status: ${res.statusCode} - Duration: ${duration}ms`);
    return originalEnd.apply(res, arguments);
  };
  
  next();
};

app.use(advancedLogger);
```

### Authentication Middleware

A JWT authentication middleware example:

```javascript
const jwt = require('jsonwebtoken');

const authenticateJWT = (req, res, next) => {
  // Get the authorization header
  const authHeader = req.headers.authorization;
  
  if (!authHeader) {
    return res.status(401).json({ error: 'Authorization header missing' });
  }
  
  // Extract the token (Bearer format)
  const token = authHeader.split(' ')[1];
  
  if (!token) {
    return res.status(401).json({ error: 'Token missing' });
  }
  
  try {
    // Verify the token
    const user = jwt.verify(token, process.env.JWT_SECRET);
    
    // Add the user object to the request for route handlers to use
    req.user = user;
    
    // Continue to the next middleware/route handler
    next();
  } catch (error) {
    if (error.name === 'TokenExpiredError') {
      return res.status(401).json({ error: 'Token expired' });
    }
    
    return res.status(403).json({ error: 'Invalid token' });
  }
};

// Apply to specific routes that need authentication
app.get('/api/profile', authenticateJWT, (req, res) => {
  // req.user is available here
  res.json({ user: req.user });
});

// Or apply to a group of routes
app.use('/api/admin', authenticateJWT);
app.use('/api/admin', adminRoutes);
```

### Role-Based Authorization Middleware

Building on authentication, you can create role-based authorization:

```javascript
const authorizeRole = (roles = []) => {
  // Convert string to array if single role
  if (typeof roles === 'string') {
    roles = [roles];
  }
  
  return (req, res, next) => {
    // Must be authenticated first
    if (!req.user) {
      return res.status(401).json({ error: 'User not authenticated' });
    }
    
    // Check if user's role is in the allowed roles
    if (!roles.includes(req.user.role)) {
      return res.status(403).json({ 
        error: 'Access denied: You do not have the required permissions' 
      });
    }
    
    next();
  };
};

// Usage:
app.get('/api/users',
  authenticateJWT,
  authorizeRole(['admin', 'manager']),
  (req, res) => {
    // Only admins and managers can access this route
    res.json({ message: 'Access granted to user list' });
  }
);
```

### Error Handling Middleware

Express.js error-handling middleware is special - it has 4 parameters instead of 3:

```javascript
const errorHandler = (err, req, res, next) => {
  // Log the error for server-side debugging
  console.error(`Error: ${err.message}`);
  console.error(err.stack);
  
  // Check error type and respond accordingly
  if (err.name === 'ValidationError') {
    return res.status(400).json({
      error: 'Validation Error',
      message: err.message,
      details: err.details || []
    });
  }
  
  if (err.name === 'UnauthorizedError') {
    return res.status(401).json({
      error: 'Authentication Error',
      message: 'Invalid token or session expired'
    });
  }
  
  // Default error response
  res.status(500).json({
    error: 'Internal Server Error',
    message: process.env.NODE_ENV === 'production' 
      ? 'Something went wrong' 
      : err.message
  });
};

// Error handler should be defined after all other middleware and routes
app.use(errorHandler);

// To use it, you can throw errors in your routes or middleware:
app.get('/example', (req, res, next) => {
  try {
    // Some operation that might fail
    throw new Error('Something went wrong');
  } catch (error) {
    // Pass the error to the error handler middleware
    next(error);
  }
});

// Or with async/await:
app.get('/async-example', async (req, res, next) => {
  try {
    // Async operation
    await someAsyncFunction();
  } catch (error) {
    next(error);
  }
});
```

## Middleware Chaining and Order of Execution

Middleware functions are executed sequentially in the order they are defined. This is often called the "middleware chain" or "middleware stack".

### Order Matters

```javascript
// Order of execution:
// 1. loggerMiddleware
// 2. authenticateJWT
// 3. Route handler
app.get('/users', 
  loggerMiddleware,
  authenticateJWT,
  (req, res) => {
    res.json({ users: [] });
  }
);

// Different order:
// 1. authenticateJWT (might end the request with a 401 response)
// 2. loggerMiddleware (only runs if authentication passes)
// 3. Route handler
app.get('/users', 
  authenticateJWT,
  loggerMiddleware,
  (req, res) => {
    res.json({ users: [] });
  }
);
```

### Conditional Next Calling

Middleware can decide whether to continue the chain or break it:

```javascript
const conditionalMiddleware = (req, res, next) => {
  if (req.query.continue === 'true') {
    console.log('Continuing to next middleware');
    next(); // Continue to next middleware
  } else {
    console.log('Ending request here');
    res.status(400).json({ message: 'Request stopped by middleware' });
    // Note: next() is not called, so the chain stops here
  }
};
```

### Waterfall Processing with Middleware

Middleware can add data to the request object for subsequent middleware to use:

```javascript
// First middleware loads user data
const loadUserData = (req, res, next) => {
  // Simulate fetching user data
  req.userData = { id: 123, name: 'John Doe' };
  next();
};

// Second middleware uses that data
const useUserData = (req, res, next) => {
  const greeting = `Hello, ${req.userData.name}!`;
  req.greeting = greeting;
  next();
};

// Route handler uses both pieces of data
app.get('/greet', loadUserData, useUserData, (req, res) => {
  res.json({ 
    greeting: req.greeting,
    userId: req.userData.id
  });
});
```

### Early Response

Middleware can end the request-response cycle early:

```javascript
const cacheMiddleware = (req, res, next) => {
  const cacheKey = req.url;
  const cachedResponse = cache.get(cacheKey);
  
  if (cachedResponse) {
    // Send the cached response and end the request cycle
    res.json(cachedResponse);
    // next() is not called, so later middleware won't run
  } else {
    // No cache hit, continue to next middleware/route handler
    next();
  }
};

app.get('/data', cacheMiddleware, (req, res) => {
  // This will only run if the data wasn't cached
  const data = fetchData();
  cache.set(req.url, data);
  res.json(data);
});
```

## Best Practices for Middleware Development

### 1. Keep Middleware Focused

Each middleware should have a single responsibility:

```javascript
// Good: Focused middleware
const validateContentType = (req, res, next) => {
  if (req.headers['content-type'] !== 'application/json') {
    return res.status(415).json({ error: 'Content type must be application/json' });
  }
  next();
};

// Bad: Doing too much
const validateAndAuthenticateAndLog = (req, res, next) => {
  // This does too many things and should be split into separate middleware
  console.log(`Request received: ${req.method} ${req.url}`);
  
  if (!req.headers.authorization) {
    return res.status(401).json({ error: 'Unauthorized' });
  }
  
  if (req.body && !req.body.name) {
    return res.status(400).json({ error: 'Name is required' });
  }
  
  next();
};
```

### 2. Handle Errors Properly

Always handle potential errors and avoid crashing the server:

```javascript
const safeMiddleware = (req, res, next) => {
  try {
    // Potentially risky operation
    const data = JSON.parse(req.body.data);
    req.parsedData = data;
    next();
  } catch (error) {
    // Handle the error instead of crashing
    res.status(400).json({ error: 'Invalid data format' });
  }
};
```

### 3. Always Call Next() (or Send a Response)

Every middleware must either call `next()` or end the response (e.g., `res.send()`, `res.json()`, `res.end()`):

```javascript
// Good: Always either calls next() or sends a response
const checkApiKey = (req, res, next) => {
  const apiKey = req.headers['x-api-key'];
  
  if (!apiKey) {
    return res.status(401).json({ error: 'API key required' });
  }
  
  if (apiKey !== process.env.API_KEY) {
    return res.status(403).json({ error: 'Invalid API key' });
  }
  
  next();
};

// Bad: Forgets to call next() in some cases
const badMiddleware = (req, res, next) => {
  if (req.query.id) {
    // Checks ID but forgets to call next()
    // The request will hang if id is present
  } else {
    next();
  }
};
```

### 4. Be Mindful of Async Operations

When using async operations in middleware, make sure to properly handle promises:

```javascript
// Bad: Doesn't handle async properly
const badAsyncMiddleware = (req, res, next) => {
  // This will continue to next middleware immediately without waiting
  database.findUser(req.params.id).then(user => {
    req.user = user;
  });
  
  next(); // Called too early!
};

// Good: Properly handling async operations
const goodAsyncMiddleware = async (req, res, next) => {
  try {
    const user = await database.findUser(req.params.id);
    req.user = user;
    next();
  } catch (error) {
    next(error);
  }
};
```

### 5. Make Middleware Configurable

Create factory functions that return middleware:

```javascript
// Configurable rate limiting middleware
const rateLimit = (options) => {
  const { windowMs = 60000, max = 100, message = 'Too many requests' } = options;
  const requests = new Map();
  
  return (req, res, next) => {
    const ip = req.ip;
    const now = Date.now();
    
    // Clean old requests
    if (!requests.has(ip)) {
      requests.set(ip, []);
    }
    
    const userRequests = requests.get(ip).filter(time => time > now - windowMs);
    requests.set(ip, userRequests);
    
    if (userRequests.length >= max) {
      return res.status(429).json({ error: message });
    }
    
    // Add current request timestamp
    userRequests.push(now);
    requests.set(ip, userRequests);
    
    next();
  };
};

// Usage:
app.use('/api', rateLimit({ windowMs: 15 * 60 * 1000, max: 100 }));
app.use('/api/sensitive', rateLimit({ windowMs: 15 * 60 * 1000, max: 20 }));
```

### 7. Test Your Middleware Thoroughly

Write unit tests for your middleware:

```javascript
// Using Jest for testing middleware
describe('authenticateJWT middleware', () => {
  it('should add user to request if token is valid', () => {
    const req = {
      headers: {
        authorization: 'Bearer valid_token'
      }
    };
    const res = {};
    const next = jest.fn();
    
    // Mock jwt.verify to return a user object
    jwt.verify = jest.fn().mockReturnValue({ id: 123, role: 'user' });
    
    authenticateJWT(req, res, next);
    
    expect(req.user).toBeDefined();
    expect(req.user.id).toBe(123);
    expect(next).toHaveBeenCalled();
  });
  
  it('should return 401 if no token provided', () => {
    const req = { headers: {} };
    const res = {
      status: jest.fn().mockReturnThis(),
      json: jest.fn()
    };
    const next = jest.fn();
    
    authenticateJWT(req, res, next);
    
    expect(res.status).toHaveBeenCalledWith(401);
    expect(res.json).toHaveBeenCalled();
    expect(next).not.toHaveBeenCalled();
  });
});
```

## Conclusion

Custom middleware is a powerful feature of Express.js that allows you to build modular, maintainable applications. By understanding the middleware function signature, execution order, and best practices, you can create sophisticated request processing pipelines that enhance your API's functionality, security, and performance.

Remember that the power of middleware comes from its simplicity and composability. Each middleware should do one thing well, and complex functionality should be built by chaining multiple middleware functions together.

As you develop your Express.js applications, consider which aspects of your request processing could be abstracted into reusable middleware, and leverage this pattern to make your code more maintainable and your APIs more robust.
